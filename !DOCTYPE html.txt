  <script>
    const buttons = [
      ['C','cls'],['+/-','sign'],['%','pct'],['÷','op'],
      ['7','num'],['8','num'],['9','num'],['×','op'],
      ['4','num'],['5','num'],['6','num'],['−','op'],
      ['1','num'],['2','num'],['3','num'],['+','op'],
      ['0','zero'],['.','dot'],['=','eq']
    ];

    let current='0', prev='', op='', fresh=false;

    function updateDisplay() {
      document.getElementById('result').textContent = current;
      document.getElementById('expr').textContent = op ? prev+' '+op : '';
    }

    function calc(a, b, o) {
      if(o==='+') return a+b;
      if(o==='−') return a-b;
      if(o==='×') return a*b;
      if(o==='÷') return b!==0 ? a/b : 0;
      return b;
    }

    function handleBtn(label, type) {
      if(type==='num') {
        if(fresh||current==='0'){current=label;fresh=false;}
        else if(current.length<12) current+=label;
      } else if(type==='zero') {
        if(fresh||current==='0'){current='0';fresh=false;}
        else if(current.length<12) current+='0';
      } else if(type==='dot') {
        if(fresh){current='0.';fresh=false;}
        else if(!current.includes('.')) current+='.';
      } else if(type==='op') {
        if(op&&!fresh){
          const r=calc(parseFloat(prev),parseFloat(current),op);
          current=String(parseFloat(r.toFixed(10)));
        }
        prev=current; op=label; fresh=true;
      } else if(type==='eq') {
        if(!op) return;
        const r=calc(parseFloat(prev),parseFloat(current),op);
        current=String(parseFloat(r.toFixed(10)));
        op=''; prev=''; fresh=true;
      } else if(type==='cls') {
        current='0'; prev=''; op=''; fresh=false;
      } else if(type==='sign') {
        current=String(parseFloat(current)*-1);
      } else if(type==='pct') {
        current=String(parseFloat(current)/100);
      }
      updateDisplay();
    }

    const grid = document.getElementById('btnGrid');
    buttons.forEach(([label, type]) => {
      const btn = document.createElement('button');
      btn.className = 'btn' +
        (type==='op' ? ' btn-op' : '') +
        (type==='eq' ? ' btn-eq' : '') +
        (type==='cls' ? ' btn-clear' : '') +
        (type==='zero' ? ' span2' : '');
      btn.textContent = label;
      btn.onclick = () => handleBtn(label, type);
      grid.appendChild(btn);
    });
  </script>
